管理员表
create table admin(
id tinyint auto_increment primary key not null,
username varchar(10) not null,
password varchar(10) not null
);
用户表
create table user(
id int auto_increment primary key not null,
username varchar(10) not null,
password varchar(20) not null,
email varchar(30) not null,
registerdate datetime not null
);
菜的分类表
create table dishcate(
id tinyint auto_increment primary key not null,
cate varchar(20) not null
);
菜表
create table dish(
id mediumint auto_increment primary key not null,
dishcate tinyint not null,
name varchar(50) not null,
price double not null,
spec tinyint default 0,
oriprice double,
reco tinyint default 0,
image varchar(50) not null,
sketch varchar(50) not null,
details text not null,
createtime datetime not null,
cartnum int default 0,
sellnum int default 0
);

评价表
create table evaluate(
id int auto_increment primary key not null,
dishid mediumint not null,
userid int not null,
star tinyint not null,
content varchar(200),
createtime datetime not null
);
订单表
create table orders(
id int auto_increment primary key not null,
userid int not null,
createtime datetime not null,
evaluated tinyint not null
);
订单详情表
create table orderdetail(
id int auto_increment primary key not null,
orderid int not null,
dishid mediumint not null,
num tinyint
);
联系电话与地址表
create table contactway(
id int auto_increment primary key not null,
userid int not null,
phone varchar(11) not null,
address varchar(50) not null
);
购物车表
create table cart(
id int auto_increment primary key not null,
userid int not null,
dishid mediumint not null,
num tinyint not null default 1,
createtime datetime not null
);
